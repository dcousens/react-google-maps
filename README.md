## Example

``` javascript
	// ...

	render: function() {
		// Melbourne
		var center = {
			lat: -37.8602828,
			lng: 145.079616
		}

		return (<GoogleMap center={center} zoom={8}>
			<InfoWindow position={center} open>
				<h1>Melbourne</h1>
			</InfoWindow>
			<Marker position={center} label="A pleasant place to live"/>
		</GoogleMap>)
	}
```
